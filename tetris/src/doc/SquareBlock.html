def initialize(row, col)
  @type = SQUARE
  @rotation_type = NO_ROTATION
  
  @blocks = [
    Block.new(row, col),
    Block.new(row, col + 1),
    Block.new(row + 1, col),
    Block.new(row + 1, col + 1)
  ]
end
