def getRotatedBlocks(row, col)
  return [
    Block.new(row, col),
    Block.new(row + 1, col),
    Block.new(row, col - 1),
    Block.new(row - 1, col - 1)
  ]
end

def getUnrotatedBlocks(row, col)
  return [
    Block.new(row, col),
    Block.new(row + 1, col),
    Block.new(row + 1, col - 1),
    Block.new(row, col + 1)
  ]
end
